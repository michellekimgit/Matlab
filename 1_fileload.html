xls_pathname = strcat(pwd,'/oneyeardata/');
MyFolderInfo = dir([xls_pathname, '*.csv']);
%MM-DDTHHMMSS_NUM_modeX.csv,
for flscnt = 1:length(MyFolderInfo)
str = MyFolderInfo(flscnt).name;
%parse name
strMM = str(1:2);
strDD = str(4:5);
strHH = str(7:8);
strMIN = str(9:10);
strSS = str(11:12);
strNUM = str(14:16);
strMode = str(22);
%
nMM = str2num(strMM);
nDD = str2num( strDD);
nHH = str2num( strHH);
nMIN = str2num( strMIN);
nSS = str2num( strSS);
nNUM = str2num( strNUM);
nMode = str2num( strMode);
%load data and store in struct
data = csvread([xls_pathname str],1,0);
%
datFile(flscnt).data = data;
datFile(flscnt).nMM = nMM;
datFile(flscnt).nDD = nDD;
datFile(flscnt).nHH = nHH;
datFile(flscnt).nMIN = nMIN;
datFile(flscnt).nSS = nSS;
datFile(flscnt).nNUM = nNUM;
datFile(flscnt).nMode = nMode;
%Compute total length of operation
datFile(flscnt).TOTALSEC = nSS + nMIN*60 + nHH*60*60 + nDD*60*60*24 + nMM*60*60*24*30;
end
[B,I] = sort([datFile(:).nNUM]);

title('Mode vs Run Num ')

hist([datFile(:).nMode]')
title('Mode Histogram for all Runs')
figure
plot([datFile(:).TOTALSEC]','+')
title('Time of Run in Total Sec vs Run Num ')
figure
plot([datFile(:).nMM]','+')
title('Month vs Run Num')
